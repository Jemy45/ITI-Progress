Deployment:
apiVersion: apps/v1
kind: Deployment
metadata:
  name: iti-deployment
  namespace: iti
spec:
  replicas: 10
  selector:
    matchLabels:
      app: nginx
  template:
    metadata:
      labels:
        app: nginx
    spec:
      containers:
      - name: nginx
        image: nginx:alpine
